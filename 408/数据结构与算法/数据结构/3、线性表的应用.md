/**
 * Definition for singly-linked list.
 * struct ListNode {
 *     int val;
 *     struct ListNode *next;
 * };
 */
struct ListNode *removeNthFromEnd(struct ListNode *head, int n) {
    // 指定一个头指针
    struct ListNode mhead;
    mhead.val=0;
    mhead.next=head;

    // 先指定快慢指针
    struct ListNode *result = &mhead;
    struct ListNode *fast = &mhead;
    //        ｜ *  ｜
    // [] 6 5 4 3 2 1
    // 快指针要比目标节点快n-1个节点，要删除它，慢指针还得在慢一拍，即快指针快一拍
    for (int i=0; i < n; i++){
        fast = fast -> next;
        if(fast == NULL) return NULL;
    }
    // 只要fast没到-1元素就同步走
    while(fast->next != NULL){
        fast = fast -> next;
        result = result->next;
    }
    // 到指定位置，删除后一个元素
    result->next = result->next->next;
    return mhead.next; // 返回头节电
}

#include <stdio.h>
#include <stdlib.h>
#define MAXSIZE 100

// 定义链表结点
typedef struct Node {
    int data;  // 存储数据
    struct Node* link;  // 指向下一个结点
} Node;

// 打印链表
void printList(Node* head) {
    Node* current = head;
    while (current != NULL) {
        printf("%d ", current->data);
        current = current->link;
    }
    printf("\n");
}

void update_linked_list(Node *head, int n){
	// 申请的bool数组用于保存数据是否存在
	bool *array = (bool *)malloc(sizeof(bool) * n);
	// 初始化
	for (int i = 0; i < n; i++)
		array[i] = false;
	// 开始遍历
	Node *current = head;
	unsigned int index;
	unsigned int data;
	while(current->link != NULL){
		// 找到下一个节点的数据data和对应的index
		data = current->link->data
		index = data > 0 ? data: -1 * data;
		if(array[index] == true){ // 前面已经存在了
			// 删除操作
			current->link = current->link->link;
		} else { // 前面不存在
			// 保存操作
			array[index] = true
			// 继续遍历
			current = current->link;
		}
	}
	// 释放array
	free(array);
}

int main() {
    // 手动拼接链表节点
    Node head = {0, NULL};
    Node node1 = {21, NULL};
    Node node2 = {-15, NULL};
    Node node3 = {-15, NULL};
    Node node4 = {-7, NULL};
    Node node5 = {-15, NULL};
    
    // 链接各个节点
    head.link = &node1;
    node1.link = &node2;
    node2.link = &node3;
    node3.link = &node4;
    node4.link = &node5;

    // 打印链表
    printf("链表内容: ");
    printList(head);
	
	update_linked_list(&head, MAXSIZE);    
	
    printf("更新后链表内容: ");
    printList(head);
    
    return 0;
}
