```c
/**
 * Definition for singly-linked list.
 * struct ListNode {
 *     int val;
 *     struct ListNode *next;
 * };
 */
struct ListNode *removeNthFromEnd(struct ListNode *head, int n) {
    // 指定一个头指针
    struct ListNode mhead;
    mhead.val=0;
    mhead.next=head;

    // 先指定快慢指针
    struct ListNode *result = &mhead;
    struct ListNode *fast = &mhead;
    //        ｜ *  ｜
    // [] 6 5 4 3 2 1
    // 快指针要比目标节点快n-1个节点，要删除它，慢指针还得在慢一拍，即快指针快一拍
    for (int i=0; i < n; i++){
        fast = fast -> next;
        if(fast == NULL) return NULL;
    }
    // 只要fast没到-1元素就同步走
    while(fast->next != NULL){
        fast = fast -> next;
        result = result->next;
    }
    // 到指定位置，删除后一个元素
    result->next = result->next->next;
    return mhead.next; // 返回头节电
}
```

```c
#define MAXSIZE 100

typedef struct Node {  
    int data;  // 存储数据  
    struct Node* link;  // 指向下一个结点  
} Node;  
  
// 打印链表  
void printList(Node* head) {  
    Node* current = head;  
    while (current != NULL) {  
        printf("%d ", current->data);  
        current = current->link;  
    }  
    printf("\n");  
}  
  
void update_linked_list(Node *head, int n){  
    // 申请的bool数组用于保存数据是否存在  
    bool *array = (bool *)malloc(sizeof(bool) * n);  
    // 初始化  
    for (int i = 0; i < n; i++)  
        array[i] = false;  
    // 开始遍历  
    Node *current = head;  
    unsigned int index;  
    int data;  
    while(current->link != NULL){  
        // 找到下一个节点的数据data和对应的index  
        data = current->link->data;  
        index = data > 0 ? data: -1 * data;  
        if(array[index] == true){ // 前面已经存在了  
            // 删除操作  
            current->link = current->link->link;  
        } else { // 前面不存在  
            // 保存操作  
            array[index] = true;  
            // 继续遍历  
            current = current->link;  
        }  
    }  
    // 释放array  
    free(array);  
}  
  
int main() {  
    // 手动拼接链表节点  
    Node head = {0, NULL};  
    Node node1 = {21, NULL};  
    Node node2 = {-15, NULL};  
    Node node3 = {-15, NULL};  
    Node node4 = {-7, NULL};  
    Node node5 = {-15, NULL};  
  
    // 链接各个节点  
    head.link = &node1;  
    node1.link = &node2;  
    node2.link = &node3;  
    node3.link = &node4;  
    node4.link = &node5;  
  
    // 打印链表  
    printf("linked list: ");  
    printList(&head);  
  
    update_linked_list(&head, MAXSIZE);  
  
    printf("updated linked list: ");  
    printList(&head);  
  
    return 0;
```

```c
void reverse_list(Node *head) {  
    // 408 有头节点的链表  
    Node *lastNode = head;  
    Node *currentNode = head->link;  
    if (currentNode == NULL) {  // 这里不判断有可能没有节点  
        return;  
    }  
    Node *nextNode = currentNode->link; // 如果有currentNode节点这个还是可能是NULL  
  
    while (nextNode != NULL) { // 只要nextNode不为NULL，也就是currentNode不是最后一个，就一直遍历  
        currentNode->link = lastNode == head ? NULL : lastNode; // 如果是第一个节点，就给他指向NULL  
        // 往后推  
        lastNode = currentNode;  
        currentNode = nextNode;  
        nextNode = currentNode -> link;  
    }  
    // 这里 nextNode == NULL, 只用更新以下currentNode的指向就可以了  
    currentNode->link = lastNode == head ? NULL : lastNode;  
    // 头节点不变，把头节点指向currentNode节点就更新完毕了  
    head->link = currentNode;  
}
```

# 删除链表中间的元素

```C
#include <stdlib.h>  
  
// 定义链表结点  
typedef struct Node {

    struct Node *link; // 指向下一个结点

    printf("\n");

void delete_mid_node(Node *head) {  
    if (head->link == NULL)  // 链表为空不更新
        return;  
        
    // 408 有头节点的链表  设定快慢指针
    Node *fast = head;  
    Node *slow = head;  
	  
	// 只要fast节点的后两个不为NULL就可以开始往后跳fast:slow = 2:1
	// fast |         |
	//      [head] [] [] [NULL] 
	// slow |
	// fast移动之前要看接下来两个是不是NULL
	// - 第一个判空保证不会导致异常
	// - 第二个判空说明当前的slow的next就是要删除的元素 
	// - 内部的判空说明
	// fast->link 不会为空 因为最开始有判空 fast->link->link->link != NULL 每次都会判断
	// while (fast->link != NULL && fast->link->link != NULL) {  // 向上取整
    while (fast->link->link != NULL && fast->link->link->link != NULL) {  // 向下取整
		fast = fast->link;        
        fast = fast->link;
        slow = slow->link;  
    }  
	  
    slow->link = slow->link->link;  
}  
  
int main() {  
    // 手动拼接链表节点  
    Node head = {0, NULL};  
    Node node1 = {1, NULL};  
    Node node2 = {2, NULL};  
    Node node3 = {3, NULL};  
    Node node4 = {4, NULL};  
    Node node5 = {5, NULL};  
    Node node6 = {6, NULL};  
  
    // 链接各个节点  
    head.link = &node1;  
    node1.link = &node2;  
    node2.link = &node3;  
    node3.link = &node4;  
    node4.link = &node5;  
    node5.link = &node6;  
  
    // 打印链表  
    printf("linked list: ");  
    printList(&head);  
  
    delete_mid_node(&head);  
  
    printf("updated linked list: ");  
    printList(&head);  
  
    return 0;
```

# 重排链表

```c
//  
// Created by shuji on 2026/3/23.  
//  
#include <stdio.h>  
#include <stdlib.h>  
  
// 定义链表结点  
typedef struct Node {  
    int data; // 存储数据  
    struct Node *link; // 指向下一个结点  
} Node;  
  
// 打印链表  
void printList(Node *head) {  
    Node *current = head;  
    while (current != NULL) {  
        printf("%d ", current->data);  
        current = current->link;  
    }  
    printf("\n");  
}  
  
void reverse_list(Node *head) {  
    // 408 有头节点的链表  
    Node *lastNode = head;  
    Node *currentNode = head->link;  
    if (currentNode == NULL) {  
        // 这里不判断有可能没有节点  
        return;  
    }  
    Node *nextNode = currentNode->link; // 如果有currentNode节点这个还是可能是NULL  
  
    while (nextNode != NULL) {  
        // 只要nextNode不为NULL，也就是currentNode不是最后一个，就一直遍历  
        currentNode->link = lastNode == head ? NULL : lastNode; // 如果是第一个节点，就给他指向NULL  
        // 往后推  
        lastNode = currentNode;  
        currentNode = nextNode;  
        nextNode = currentNode->link;  
    }  
    // 这里 nextNode == NULL, 只用更新以下currentNode的指向就可以了  
    currentNode->link = lastNode == head ? NULL : lastNode;  
    // 头节点不变，把头节点指向currentNode节点就更新完毕了  
    head->link = currentNode;  
}  
  
void resorted_linked_list(Node *head) {  
    // head[][][]  
    if (head->link == NULL || head->link->link == NULL || head->link->link->link == NULL) // 1 ~ 2 个不用换  
        return;  
    // 408 有头节点的链表  
    Node *fast = head;  
    Node *slow = head;  
    while (fast->link != NULL && fast->link->link != NULL) {  
        fast = fast->link;  
        fast = fast->link;  
        slow = slow->link;  
    }  
    Node list2;  
    list2.data = 0;  
    list2.link = slow->link;  
    slow->link = NULL;  
    reverse_list(&list2);  
  
    // 3个元素及以上  
    //   head []   [] -> NULL  
    //         | | |    // &list2 []   [] -> NULL    printList(head);  
    printList(&list2);  
    Node *h1 = head->link;  
    Node *h2 = head->link->link;  
    Node *l1 = list2.link;  
    Node *l2 = list2.link->link;  
  
    h1->link = l1;  
    l1->link = h2;  
  
    while (h2 != NULL && l2 != NULL) {  
        h1 = h2;  
        h2 = h2->link;  
        l1 = l2;  
        l2 = l2->link;  
        h1->link = l1;  
        l1->link = h2;  
    }  
}  
  
int main() {  
    // 手动拼接链表节点  
    Node node10 = {10, NULL};  
    Node node9 = {9, &node10};  
    Node node8 = {8, &node9};  
    Node node7 = {7, &node8};  
    Node node6 = {6, &node7};  
    Node node5 = {5, &node6};  
    Node node4 = {4, &node5};  
    Node node3 = {3, &node4};  
    Node node2 = {2, &node3};  
    Node node1 = {1, &node2};  
    Node head = {0, &node1};  
  
    // 打印链表  
    printf("linked list: ");  
    printList(&head);  
  
    resorted_linked_list(&head);  
  
    printf("updated linked list: ");  
    printList(&head);  
  
    return 0;  
}
```
