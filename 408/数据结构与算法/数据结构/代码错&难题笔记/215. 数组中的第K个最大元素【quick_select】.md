# [215. 数组中的第K个最大元素](https://leetcode.cn/problems/kth-largest-element-in-an-array/)

```c
int fastsort(int* nums, int numsSize) {
	// 获取到第一个位置的元素，和头尾节点的方法内index；
	// 方法至少需要一个元素
	int x = rand() % numsSize;
	int tmp = nums[x];
	nums[x] = nums[0];
	int left = 0;
	int right = numsSize - 1;
	// 1个元素：left==right && 5个元素：left = 0 & right = 4 
	int move = 0;
	// 1代表右，0代表左
	while(left != right) {
		if(move == 0) {
			// 左边空时
			if(tmp > nums[right]) {
				// 右边元素位置错误
				// 移到左边
				nums[left] = nums[right];
				// 左index + 1
				left ++;
				move = 1;
			} else {
				// 如果正确
				// 右index - 1
				right --;
			}
		} else {
			// 右边空时
			if(tmp < nums[left]) {
				// 右边元素位置错误
				// 移到右边
				nums[right] = nums[left];
				// 右index - 1
				right --;
				move = 0;
			} else {
				// 如果正确
				// 左index + 1
				left ++;
			}
		}
	}
	nums[left] = tmp;
	return left;
}
int findKthLargest(int* nums, int numsSize, int k) {
	if (numsSize == 0 || numsSize < k || nums == NULL) return 0;
	if (numsSize == 1 && k == 1) return nums[0];
	int start = 0;
	int end = numsSize - 1;
	int endN = -1;
	// 代表目前选中的元素是原来数组的第几大的元素
	int index = -1;
	while(endN != k) {
		// 首先，我们对数组做一次快排，就能得到一个元素的相对位置
		index = fastsort(nums + start, end + 1 - start);
		// index 是相对位置
		endN = numsSize - (start + index);
		// index是倒数第几个元素
		if (endN < k) {
			// 然后如果这个元素在k的左边我们就把左边快排
			end = start + index - 1;
		} else if (endN > k) {
			// 然后如果这个元素在k的右边我们就把右边快排
			start = start + index + 1;
		}
	}
	return nums[start + index];
}
```


# 标准解法 快速选择
```c
int quickselect(int* nums, int l, int r, int k) {
	if (l == r) return nums[k];
	int partition = nums[l];
	int i = l - 1;
	int j = r + 1;
	int tmp;

	while(i < j) {
		do i++;	while (nums[i] < partition);
		do j--; while (nums[j] > partition);
		if (i < j) {
			tmp  = nums[i];
			nums[i] = nums[j];
			nums[j] = tmp;
		}
	}
	quickselect(nums, l, j, k); 
	quickselect(nums, j + 1, r, k);
}

int findKthLargest(int* nums, int numsSize, int k) {
	return quickselect(nums, 0, numsSize - 1, numsSize - k);
}
```
