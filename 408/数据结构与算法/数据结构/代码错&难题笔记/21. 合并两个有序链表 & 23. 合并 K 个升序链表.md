# [21. 合并两个有序链表](https://leetcode.cn/problems/merge-two-sorted-lists/)

![[images/Pasted image 20260329200527.png|441]]

```c
/**
 * Definition for singly-linked list.
 * struct ListNode {
 *     int val;
 *     struct ListNode *next;
 * };
 */
struct ListNode* mergeTwoLists(struct ListNode* list1, struct ListNode* list2) {
    // 排除空节点的情况
    if (list1 == NULL && list2 == NULL) return NULL;
    if (list1 == NULL || list2 == NULL) return list1 == NULL ? list2 : list1;
    
    // 头阶点保存
    struct ListNode* head = list1->val <= list2->val ? list1: list2; // 输出头阶点
    
    
    struct ListNode* prev = NULL;
    struct ListNode* current = list1;
    struct ListNode* another = list2;
    while(current != NULL && another != NULL){
        if(current->val <= another->val){
            if (prev != NULL) {
                prev->next = current;
            }
            prev = current;
            current = current->next;
            if (current == NULL) {
                prev -> next = another;
            }
        } else {
            if (prev != NULL) prev->next = another;
            prev = another;
            another = another->next;
            if (another == NULL) {
                prev -> next = current;
            }
        }
    }
    return head;
}
```

# [23. 合并 K 个升序链表](https://leetcode.cn/problems/merge-k-sorted-lists/)
第一种：合并两个然后复用代码
```c
struct ListNode* mergeKLists(struct ListNode** lists, int listsSize) {
    if (listsSize == 0) return NULL;
    if (listsSize == 1) return lists[0];
    struct ListNode* head = lists[0];
    for(int i = 1; i < listsSize; i++){
        head = mergeTwoLists(head, lists[i]); // mergeTwoLists 在上面
    }
    return head;
}
```

第二种：优化代码，开辟一个数组存放多个指针；每次都找到接下来要指向哪里(但看起来时间复杂度好像更差了)
```c
/**
 * Definition for singly-linked list.
 * struct ListNode {
 *     int val;
 *     struct ListNode *next;
 * };
 */
int find_min_data_index(int *data, int listsSize){
    int min = 0;
    for (int i = 1; i < listsSize; i++){
        min = data[min] < data[i] ? min : i;
    }
    return min;
}
struct ListNode* mergeKLists(struct ListNode** lists, int listsSize) {
    if (listsSize == 0) return NULL;
    if (listsSize == 1) return lists[0];
    struct ListNode* ptr[listsSize];
    int data[listsSize];
    for (int i = 0; i < listsSize; i++){
        ptr[i] = lists[i];
        data[i] = lists[i] == NULL ? INT_MAX : lists[i]->val;
    }
    struct ListNode* head = NULL;
    struct ListNode* prev = NULL;
    while(!(prev == NULL && head != NULL)){
        int min_index = find_min_data_index(data, listsSize);
        if(data[min_index] == INT_MAX){
            prev = NULL;
            break;
        }
        if (prev != NULL) {
            prev->next = ptr[min_index];
        } else {
            head = ptr[min_index];
        }
        prev = ptr[min_index];
        ptr[min_index] = ptr[min_index] -> next;
        data[min_index] = ptr[min_index] == NULL? INT_MAX : ptr[min_index]->val;
    }
    return head;
}
```
第三种：依旧用合并两个链表的代码，可是分治
```c
struct ListNode* mergeKLists(struct ListNode** lists, int listsSize) {
    if (listsSize == 0) return NULL;
    if (listsSize == 1) return lists[0];
    int arr_size = listsSize;
    while(arr_size != 1){
        for (int i = 0; i < (arr_size + 1) / 2 ; i++){
            lists[i] = mergeTwoLists(lists[i*2], ((i*2+1) == arr_size) ? NULL: lists[i*2+1]);
        }
        arr_size = (arr_size + 1) / 2;
    }
    return lists[0];
}
```

根据这个代码简单总结可知，如果是把有限个问题扩展到若干个，往往是把问题用分治归并解决。
