double findMedianSortedArrays(int* nums1, int nums1Size, int* nums2, int nums2Size) {
	// 交换两个数组次序，让大的数据在nums2
    int * arr = NULL;
    int size = -1;
    if (nums1Size > nums2Size){
        arr = nums1;
        nums1 = nums2;
        nums2 = arr;
        size = nums1Size;
        nums1Size = nums2Size;
        nums2Size = size;
    }
    
    
    int half = (nums1Size + nums2Size + 1) / 2; // 左边需要有多少个元素
    
	// 对num1进行二分查找
    int left = 0;
    int right = nums1Size;
    int cut1, cut2;
    while (left < right){
        cut1 = left + (right - left + 1) / 2; // mid_index
        cut2 = half - cut1;
        if(nums1[cut1-1] > nums2[cut2]) {
            right = cut1 - 1;
			// 太靠右了往左更新
        } else {
            left = cut1;
			// 太靠左了往右更新
        }
    }
    cut1 = left;
    cut2 = half - left;
    
    // 正确线的左右数据
    int num1_left = cut1 == 0 ? INT_MIN : nums1[cut1-1];
    int num1_right = cut1 == nums1Size ? INT_MAX : nums1[cut1];
    int num2_left = cut2 == 0? INT_MIN : nums2[cut2-1];
    int num2_right = cut2 == nums2Size? INT_MAX : nums2[cut2];
    
    // 单双区分
    if ((nums1Size + nums2Size) % 2 == 1){
        return (double) (num1_left < num2_left ? num2_left : num1_left);
    } else {
        return (double) ((num1_left < num2_left ? num2_left : num1_left)
        + (num1_right < num2_right ? num1_right : num2_right)) / 2;
    }
}
