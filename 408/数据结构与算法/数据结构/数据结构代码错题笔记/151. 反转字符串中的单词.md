# [151. 反转字符串中的单词](https://leetcode.cn/problems/reverse-words-in-a-string/)

写法2: 新数组：空间O(n),时间O(n)
写法流程：
1、先找到两侧的边界，然后倒序进行读取，读完一个单词（读到空格）就拷贝数据。
2、一直把数据拷贝完成即止。
```c
char* reverseWords(char* s) {
	int left = 0;
	int right = 0;
	while(s[right] != '\0') {
		right++;
	}
	right--;
	// 定位到最后一个字符
	while(s[left] == ' ') {
		left++;
	}
	// 定位到第一个字母
	while(s[right] == ' ') {
		right--;
	}
	// 定位到最后一个字母 // 's...d'
	if (left > right) {
		s[0] = '\0'
		return s; 
	}
	// 申请数组
	char* arr = (char *)malloc(sizeof(char)*(right-left+2));
	arr[right-left+1]='\0'; // 给最后一位置0
	
	int index = right;
	int copy_start = left;
	int writer = 0;
	int add_space_count = 0;
	// 倒着检测是否碰到空格了，碰到空格了就顺着把单词复制进去，然后加上空格，如果已经上次写的是空格，就不再写入空格，并且把空格计数，
	while(index != left) {
		if(s[index] == ' ') {
			int i = index + 1;
			while (s[i] != ' ' && s[i] != '\0') {
				arr[writer++] = s[i++];
			}
			if(arr[writer-1] != ' ') arr[writer++] = ' '; else add_space_count++;
		}
		index --;
	}
	
	// 把最后的数据写进去
	while (s[copy_start] != ' ' && s[copy_start] != '\0') {
		arr[writer++] = s[copy_start++];
	}
	
	// 通过多的空格，修改 '\0' 的位置
	arr[right-left+1-add_space_count]='\0';
	return arr;
}
```

写法1-反转两次：空间O(1),时间O(n)
写法流程：
1、反转两次的要求是反转的数据要是干净的 → 先做规格化。trim & delete_mid_space
2、先反转内容一次，再根据空格把单词反转回来。
写法要点：
1、不要想着一次写完，多分开写函数更清晰也更可复用（基本的违法值判断要有）。
2、先写注释，在写代码。// 先理清思路，再动笔写。
3、写完注意是否';'和有没有把赋值写成'\=\='。
4、写了标记变量就要判断是否需要修改这个变量的值。
```c
# include <stdio.h>

void reverse_words(char* s, int n){  
    if(n < 2) return;  
    char tmp = ' ';  
    for (int i = 0; i < n / 2; i++){  
        tmp = s[n - 1 - i];  
        s[n - 1 - i] = s[i];  
        s[i] = tmp;  
    }  
}  
  
char* trim(char* s){  
    int left = 0;  
    int right = 0;  
    while(s[right] != '\0') right++; // right最终落在 '\0'    while(s[left] == ' ') left++;  
  
    if (s[left] == '\0') {// 如果left这里过了，说明最少有一个非空格字符        return s + left;  
    }  
    right --; // 从right - 1 开始遍历    while(s[right] == ' ') right--;  
  
    s[right + 1] = '\0';  
    // 找到边界，然后返回字符    return s + left;  
}  
  
void delete_mid_space(char* s){  
    int last_is_space = 0;  
    int skip_space_num = 0;  
    int index = 0;  
    int writer = 0;  
  
    while(s[index] != '\0'){  
        if(s[index] != ' '){  
            // 如果不是空格，挪动到相应位置            s[writer] = s[index];  
            index ++;  
            writer ++;  
            last_is_space = 0;  
        } else {  
            if (last_is_space == 0){  
                // 如果是第一个空格，那么就写一次空格                s[writer] = ' ';  
                index ++;  
                writer ++;  
                last_is_space = 1;  
                continue;  
            }  
            // 如果是第1+个空格，只往后挪，跳过这个元素            index ++;  
            skip_space_num ++;  
            last_is_space = 1;  
        }  
  
    }  
    // 这时 s[index] == '\0'  
    // 跳过 n 个元素，我要把 '\0' 往前搬 n 个元素    s[index - skip_space_num] = '\0';  
}  
  
char* reverseWords(char* s) {  
    s = trim(s); // 删除两侧的空格    delete_mid_space(s); // 删除中间多余的空格    int cnt = 0;  
    while(s[cnt] != '\0') cnt++;  
  
    reverse_words(s, cnt); // 反转整个单词    int word_first = -1;  
    int index = 0;  
    while (s[index] != '\0'){  
        if(s[index] == ' ' && word_first != -1){ // 扫描到空格 处理前面的单词            reverse_words(s + word_first, index - word_first);  
            index ++;  
            word_first = -1;  
        } else if (word_first == -1){ // 扫描到单词的第一个字母            word_first = index;  
            index ++;  
        } else { // 扫描到单词的非第一个字母            index ++;  
        }  
    }  
    if(word_first != -1) {  
        reverse_words(s + word_first, index - word_first);  
    }  
    return s;  
}  
  
int main() {  
    printf("\"");  
    char s[] = "  hello   world  ";  
    printf(reverseWords(s));  
    printf("\"");  
    return 0;  
}
```
