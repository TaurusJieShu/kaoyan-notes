> 初始定义，这里可以作为最简单的结构进行加入后续的方法。
```c
#include<stdio.h>  
  
typedef int ElementType;  
  
typedef struct node {  
    ElementType data;  
    struct node *next;  
} Node;  
  
void init_Node(Node *node) {  
    node->data = 0;  
    node->next = NULL;  
}  

int main() {  
    Node head;  
    init_Node(&head);    
    return 0;  
}
```

单链表实现

```c
//  
// Created by shuji on 2026/3/22.  
//  
  
#include<stdio.h>  
  
typedef int ElementType;  
  
typedef struct node {  
    ElementType data;  
    struct node *next;  
} Node;  
  
void init_Node(Node *node) {  
    node->data = 0;  
    node->next = NULL;  
}  
  
int insert_Node_to_head(Node *head, Node *node, ElementType data) {  
    // 准备好节点  
    init_Node(node);  
    node->data = data;  
  
    // 1、把head的后一个节点地址给node节点  
    node->next = head->next;  
    // 2、把head的next节点指向node节点  
    head->next = node;  
    // 返回成功  
    return 1;  
}  
  
int insert_Node_to_end(Node *head, Node *node, ElementType data) {  
    // 准备好节点  
    init_Node(node);  
    node->data = data;  
  
    Node *current = head;  
    // 找到最后的节点  
    while (current->next != NULL) {  
        current = current->next;  
    }  
    // current就是最后的节点 current->next==NULL，由于已经初始化过了，则插入成功  
    current->next = node;  
  
    // 返回成功  
    return 1;  
}  
  
int insert_Node_by_pos(Node *head, Node *node, ElementType data, int pos) {  
    if (pos < 1) return 0;  
    // pos 是head节点后的第一个节点为1插入就是替换该点的数据把该点的数据往后挪一格(对数组而言);  
    // 准备好节点  
    init_Node(node);  
    node->data = data;  
  
    Node *current = head;  
  
    // 找到目标节点的前一个节点，如果是替换第2个节点，也就是我从head节点指针挪动1次  
    for (int i = 0; i < pos - 1; i++) {  
        if (current == NULL) return 0;  
        current = current->next;  
    }  
    // 循环完了之后，把新节点插到current节点的后面  
    node->next = current->next;  
    current->next = node;  
  
    // 返回成功  
    return 1;  
}  
  
int delete_node_by_pos(Node *head, int pos, Node *out_node) {  
    // pos不合法  
    if (pos < 1) return 0;  
  
    // 找到目标节点的前一个节点，如果我要删除第1个节点，我就要找到第0个节点  
    Node *current = head;  
    for (int i = 0; i < pos - 1; i++) {  
        if (current == NULL) return 0; // 删除的节点超过目标  
        current = current->next;  
    }  
    if (current->next == NULL) return 0; // 目标节点前一个节点是尾节点 -> 节点不存在  
  
    // 后面不反回节点 -> 这里改成返回节点  
    out_node = current->next;  
    // 已经找到了目标节点，节点形式是 current->next == 目标节点  
    // 找到目标节点下一个节点的地址（可能是NULL）  
    // 把当前节点指向下一个节点的地址  
    current->next = current->next->next;  
  
    // free 目标节点（这里没有用动态的暂时不释放）# todo  
    return 1;  
}  
  
void print_Nodes(Node *head) {  
    printf("----print all nodes-----------------\n");  
    if (head->next == NULL) {  
        printf("Node NULL!");  
    }  
    Node *current = head->next;  
    // 这里不考虑头节点所以要用头节点的下一个节点开始  
    while (current != NULL) {  
        printf("%d -> ", current->data);  
        current = current->next;  
    }  
    printf("\n-------------------------\n");  
}  
  
  
int main() {  
    Node head;  
    init_Node(&head);  
    Node data1, data2, data3, data4, data5;  
    Node data6, data7, data8, data9, data10;  
  
    insert_Node_to_head(&head, &data1, 5);  
    insert_Node_to_head(&head, &data2, 4);  
    insert_Node_to_head(&head, &data3, 3);  
    insert_Node_to_head(&head, &data4, 2);  
    insert_Node_to_head(&head, &data5, 1);  
  
    insert_Node_to_end(&head, &data6, 6);  
    insert_Node_to_end(&head, &data7, 7);  
    insert_Node_to_end(&head, &data8, 8);  
    insert_Node_to_end(&head, &data9, 9);  
    insert_Node_to_end(&head, &data10, 10);  
    print_Nodes(&head);  
    return 0;  
  
    Node node11;  
    int success = insert_Node_by_pos(&head, &node11, 100, 200);  
    printf(success ? "insert success" : "insert fail");  
    printf("\n");  
    print_Nodes(&head);  
    Node *out_node;  
    int success1 = delete_node_by_pos(&head, 10, out_node);  
    printf(success1 ? "delete success" : "delete fail");  
    printf("\n");  
    print_Nodes(&head);  
  
  
    return 0;  
}
```
