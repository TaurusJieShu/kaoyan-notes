> 
> 这里全都是各种方法实现，除了insert的方法有嵌套函数，其他的都是基于 SeqList_dynamic来的，这里解释一下命名：
> - 除了SeqList_dynamic是为了跟SeqList区分开，加了一个_dynamic，其他的都是正常的命名规则SeqList是顺序表，SeqList_dynamic是堆存储的顺序表（需要malloc和free）
> 
```c
#include <stdio.h>  
#include <stdlib.h>  
  
#define MAXSIZE 100  
  
typedef int ElementType;  
  
  
typedef struct {  
    ElementType data[MAXSIZE];  
    int length;  
} SeqList;  
  
// 初始化直接声明的顺序表
void init_seq_list(SeqList *seq_list) {  
    seq_list->length = 0;  
}  
  
typedef struct {  
    ElementType *data;  
    int length;  
} SeqList_dynamic;  
  
// 初始化malloc的动态顺序表
void init_seq_list_dynamic(SeqList_dynamic *seq_list) {  
    seq_list->data = (ElementType *) malloc(MAXSIZE * sizeof(ElementType));  
    seq_list->length = 0;  
}  

// 销毁malloc的动态顺序表
void destroy_seq_list_dynamic(SeqList_dynamic *seq_list) {  
    free(seq_list->data);  
}  
  
int insert_element_to_end(SeqList_dynamic *seqList, ElementType element) {  
    if (seqList->length >= MAXSIZE) {  
        // 数组越界判定  
        printf("Fail Insert %d to SeqList\n", element);  
        return 0;  
    }  
    seqList->data[seqList->length] = element; // 添加内容  
    seqList->length++; // 更新属性  
    printf("Success Insert %d to SeqList\n", element);  
    return 1; // 返回成功  
}  

// 打印动态列表
void print_SeqList_dynamic(const SeqList_dynamic *const seqList) {  
    if (seqList->length == 0) return;  
    for (int i = 0; i < seqList->length; i++) {  
        printf("ElementType Data: %d\n", seqList->data[i]);  
    }  
}  

// 打印列表
void print_SeqList(const SeqList *const seqList) {  
    if (seqList->length == 0) return;  
    for (int i = 0; i < seqList->length; i++) {  
        printf("ElementType Data: %d\n", seqList->data[i]);  
    }  
}  
  
// 按照位置删除元素
int delete_element_by_pos(SeqList_dynamic *seqList, int pos, ElementType *out_element) {  
    if (seqList->length <= pos) {  
        // 数组越界判定  
        printf("Fail Delete %d from SeqList\n", pos);  
        return 0;  
    }  
    *out_element = seqList->data[pos];  
    for (int i = pos + 1; i < seqList->length; i++) {  
        seqList->data[i - 1] = seqList->data[i];  
    }  
    seqList->length--; // 更新属性  
    printf("Success Delete %d to SeqList\n", *out_element);  
    return 1; // 返回成功  
}  
  
// 按照位置插入元素
int insert_element_by_pos(SeqList_dynamic *seqList, ElementType element, int pos) {  
    if (seqList->length >= MAXSIZE || pos >= MAXSIZE) {  
        // 数组越界判定  
        printf("Fail Insert %d to SeqList\n", element);  
        return 0;  
    }  
    if (seqList->length <= pos) {  
        // 变成尾插  
        insert_element_to_end(seqList, element);  
        return 1;  
    }  
    for (int i = seqList->length; i > pos; i--) {  
        seqList->data[i] = seqList->data[i - 1];  
    }  
    seqList->data[pos] = element;  
    seqList->length++; // 更新属性  
    printf("Success Insert %d to SeqList\n", element);  
    return 1; // 返回成功  
}  
  
// 按照内容找到元素
int find_element_by_context(const SeqList_dynamic * const seqList, ElementType element, int *out_pos) {  
    if (seqList->length == 0) {  
        printf("No Data");  
        return 0;  
    }  
    for (int i = 0; i < seqList->length; i++) {  
        if (seqList->data[i] == element) {  
            *out_pos = i;  
            return 1;  
        }  
    }  
    return 0;  
}  
  
int main(void) {  
    SeqList_dynamic *seq_list = (SeqList_dynamic *) malloc(sizeof(SeqList_dynamic));  
    init_seq_list_dynamic(seq_list);  
  
    printf("--------------尾部插入-----------------\n");  
    insert_element_to_end(seq_list, 0);  
    insert_element_to_end(seq_list, 1);  
    insert_element_to_end(seq_list, 2);  
    insert_element_to_end(seq_list, 3);  
    insert_element_to_end(seq_list, 4);  
    insert_element_to_end(seq_list, 5);  
    printf("--------------位置插入-----------------\n");  
    insert_element_by_pos(seq_list, 10, 6);  
    process_element(seq_list);  
    printf("--------------位置删除-----------------\n");  
    ElementType remove_element;  
    delete_element_by_pos(seq_list, 2, &remove_element);  
    print_SeqList_dynamic(seq_list); // 遍历并打印list
    printf("--------------找出元素-----------------\n");  
    int *find_int;  
    ElementType find_element = 20;  
    int success = find_element_by_context(seq_list, find_element, &find_int);  
    if (success) {  
        printf("Find data: %d, pos_index: %d \n", find_element, find_int);  
        return 1;  
    }  
    printf("Can't Find data %d", find_element);  
	  
	  
	// 调用方法先销毁内部结构
    destroy_seq_list_dynamic(seq_list);  
    // 再调用方法销毁列表（其实这个列表没必要malloc，因为就占两个int）
    free(seq_list);  
    return 0;  
}
```
