# [973. 最接近原点的 K 个点](https://leetcode.cn/problems/k-closest-points-to-origin/)

```c
/**
 * Return an array of arrays of size *returnSize.
 * The sizes of the arrays are returned as *returnColumnSizes array.
 * Note: Both returned array and *columnSizes array must be malloced, assume caller calls free().
 */
void swap(int* point1, int* point2) {
	int t = point1[0];
	point1[0] = point2[0];
	point2[0] = t;
	t = point1[1];
	point1[1] = point2[1];
	point2[1] = t;
}
int get_distance(int* nums) {
	return nums[0]*nums[0] + nums[1]*nums[1];
}
void quick_select(int** nums, int l, int r, int k) {
	if(l == r) return;
	int compared_num = get_distance(nums[l]);
	int i = l - 1;
	int j = r + 1;
	while(i < j) {
		do i++;
		while(get_distance(nums[i]) < compared_num);
		do j--;
		while(get_distance(nums[j]) > compared_num);
		if (i < j) swap(nums[i], nums[j]);
	}
	if (j <= k) quick_select(nums, j+1, r, k); else quick_select(nums, l, j, k);
}
int** kClosest(int** points, int pointsSize, int* pointsColSize, int k, int* returnSize, int** returnColumnSizes) {
	quick_select(points, 0, pointsSize - 1, k - 1);
	int** arr = (int**)malloc(k*sizeof(int)*2);
	for (int i = 0; i < k; i++) {
		arr[i] = points[i];
	}
	*returnSize = k;
	*returnColumnSizes = (int*)malloc(sizeof(int) * k);
	for (int i = 0; i < k; i++) {
		(*returnColumnSizes)[i] = 2;
	}
	return arr;
}
```
