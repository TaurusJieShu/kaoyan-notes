```c
int evalRPN(char** tokens, int tokensSize) {
    int stack[tokensSize];
    int top = -1;
    for (int i = 0; i< tokensSize; i++){
        if((strlen(tokens[i]) > 1) || (tokens[i][0] >= '0' && tokens[i][0] <= '9')){
            stack[++top] = atoi(tokens[i]);
            continue;    
        }
        int op_a = stack[top--];
        int op_b = stack[top--];
        if (strcmp(tokens[i], "+") == 0){
            stack[++top] = op_b + op_a;
        }
        if (strcmp(tokens[i], "-") == 0){
            stack[++top] = op_b - op_a;
        }
        if (strcmp(tokens[i], "*") == 0){
            stack[++top] = op_b * op_a;
        }
        if (strcmp(tokens[i], "/") == 0){
            stack[++top] = (int)(op_b / op_a);
        }
    }
    return stack[0];
}
```

## 算法思路

```
初始化空栈
遍历每个 token：
    if token 是数字：
        压入栈
    else token 是运算符：
        弹出右操作数 → b
        弹出左操作数 → a
        计算 a op b
        结果压入栈
返回栈顶元素
```
