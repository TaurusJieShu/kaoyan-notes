# [61. 旋转链表](https://leetcode.cn/problems/rotate-list/)

```c
//  
// Created by shuji on 2026/3/24.  
//  
/**  
 * Definition for singly-linked list. * struct ListNode { *     int val; *     struct ListNode *next; * }; */
struct ListNode *rotateRight(struct ListNode *head, int k) {  
    // 排除0个，1个元素，k不合法的情况  
    if (head == NULL || head->next == NULL || k <= 0) {  
        return head;  
    }  
  
    // 格式化 k 值  
    struct ListNode *current = head;  
    int cnt = 0;  
    while (current != NULL) {  
        cnt++;  
        current = current->next;  
    }  
    k = k % cnt;  
    if (k == 0) {  
        return head;  
    }
  
    struct ListNode *prev = NULL;  
    current = head;  
    struct ListNode *prev_fast = NULL;  
    struct ListNode *fast = head;  
  
  
    for (int i = 0; i < k; i++) {  
        fast = fast->next;  
    }  
  
    while (fast != NULL) {  
        prev_fast = fast;  
        fast = fast->next;  
        prev = current;  
        current = current->next;  
    }  
  
    prev->next = NULL;  
    prev_fast->next = head;  
    return current;  
}
```
# [876. 链表的中间结点](https://leetcode.cn/problems/middle-of-the-linked-list/)
```c
/**  
* Definition for singly-linked list.  
 * struct ListNode { *     int val; *     struct ListNode *next; * }; */
   
struct ListNode* middleNode(struct ListNode* head) {  
    struct ListNode* slow = head;  
    struct ListNode* fast = head;  
  
    while(fast!=NULL && fast->next!=NULL){  
        slow = slow -> next;  
        fast = fast -> next;  
        fast = fast -> next;  
    }  
  
    return slow;  
}
```

# [141. 环形链表](https://leetcode.cn/problems/linked-list-cycle/)

```c
/**  
* Definition for singly-linked list.  
 * struct ListNode { *     int val; *     struct ListNode *next; * }; */
bool hasCycle(struct ListNode *head) {  
    struct ListNode *fast = head;  
    struct ListNode *slow = head;  
  
    while (fast != NULL && fast -> next != NULL){  
        fast = fast->next;  
        if(fast == slow) return true;  
        fast = fast->next;  
        if(fast == slow) return true;  
  
        slow = slow->next;  
    }  
    return false;  
}
```

# [189. 轮转数组](https://leetcode.cn/problems/rotate-array/)

```c
//

//  
void rotate(int *nums, int numsSize, int k) {  
    k = k % numsSize;  
    if (k == 0) return;  
    int a[k];  
    int index = 0;  
    for (int i = 0; i < k; i++) {  
        a[i] = nums[numsSize - k + i];  
    }  
    for (int i = numsSize - k - 1; i >= 0; i--) {  
        nums[i + k] = nums[i];  
    }  
    for (int i = 0; i < k; i++) {  
        nums[i] = a[i];  
    }  
}
```
> 
> 这里其实很有意思的一点是，我们通常希望：“**数据一次就到我们需要的位置**”，👆的思想就是这样。
> 但其实有个很重要的思想就是，如果我们**不用一次完成**，先近一点，在近一点，再近一点，或许就能完成，而👇的思想就是非一次性完成，而且代码更有可读性。
> 这类题不一定要盯着：
> - 这个数最后去哪
> -那个数最后去哪
> 更重要的是看：
> - 整体结构能不能拆成几段
> - 这些段能不能通过某种标准操作重组
> - 有没有一种“中间状态”更容易实现
>

```c
    k = k % numsSize;  
    if (numsSize < 2 || k == 0) return;  
    reverse_array(nums, numsSize);  
    reverse_array(nums, k);  
    reverse_array(nums + k, numsSize - k);  
}
```

# [704. 二分查找](https://leetcode.cn/problems/binary-search/)
```c
int search(int* nums, int numsSize, int target) {
    int left = 0;
    int right = numsSize - 1;
    int cut;
    while (left < right){
        cut = left + (right - left + 1) / 2;
        if (nums[cut] == target) return cut;
        else if (nums[cut] > target){
            right = cut - 1;
        } else {
            left = cut;
        }
    }
    cut = left;
    if (nums[cut] != target){
        return -1;
    }
    return cut;
}
```

# [160. 相交链表](https://leetcode.cn/problems/intersection-of-two-linked-lists/)
```c
/**
 * Definition for singly-linked list.
 * struct ListNode {
 *     int val;
 *     struct ListNode *next;
 * };
 */
struct ListNode *getIntersectionNode(struct ListNode *headA, struct ListNode *headB) {
    int cnt1 = 0;
    int cnt2 = 0;
    struct ListNode *head_A = headA;
    struct ListNode *head_B = headB;
    while(head_A != NULL){
        head_A = head_A -> next;
        cnt1++;
    }
    while(head_B != NULL){
        head_B = head_B -> next;
        cnt2++;
    }
    head_A = headA;
    head_B = headB;
    struct ListNode *longH = cnt1 <= cnt2 ? head_B: head_A;
    struct ListNode *shortH = cnt1 <= cnt2 ? head_A: head_B;
    int cnt = cnt1 <= cnt2 ? cnt2 - cnt1: cnt1 - cnt2;
    for(int i=0; i < cnt; i++){
        longH = longH -> next;
    }
    while(longH != shortH){
        longH = longH -> next;
        shortH = shortH -> next;
    }
    return longH;
}
```

```c
/**
 * Definition for singly-linked list.
 * struct ListNode {
 *     int val;
 *     struct ListNode *next;
 * };
 */
bool hasCycle(struct ListNode *head) {
    if (head == NULL) return false;
    struct ListNode *fast = head;
    struct ListNode *slow = head;
    while(fast->next != NULL && fast->next->next != NULL){
        fast = fast->next;
        if (fast == slow) return true;
        fast = fast->next;
        if (fast == slow) return true;
        slow = slow->next;
    }
    return false;
}
```

# [229. 多数元素 II](https://leetcode.cn/problems/majority-element-ii/)
```c
/**
 * Note: The returned array must be malloced, assume caller calls free().
 */
int* majorityElement(int* nums, int numsSize, int* returnSize) {
    int major1, major2; // 建立n个阵营
    int cnt1 = 0, cnt2 = 0;
    for (int i = 0; i < numsSize; i++){
        int cur = nums[i]; // 当前元素
// 如果阵营人数为 0 就需要被占用
        if (cnt1 == 0 && cur != major2){
// if (cnt1 == 0){ 这样不能避免当前值是另一个阵营的时候，优先应该归到隔壁去
            major1 = cur;
        } else if (cnt2 == 0 && cur != major1){
            major2 = cur;
        }
// 如果是AB阵容，就归到AB去，如果不是，那2家都牺牲一名士兵
        if(major1 == cur){
            cnt1++;
        } else if(major2 == cur){
            cnt2++;
        } else {
// 避免cnt 小于 0, 至多等于0
            cnt1 -= (cnt1 == 0)? 0:1; // 如果另一个阵营本来就没有元素了，那么就不扣了
            cnt2 -= (cnt2 == 0)? 0:1;
        }
// 由于两个阵营的人数都超过了总人数的1/3，所以就算所有非两人阵营的人，都来跟这两个阵营的人抵消，都能得到最终的两个阵营
    }
// 这里如果扩展到n个阵营，num/n 也是可以用这么一个方法。
    cnt1 = 0;
    cnt2 = 0;
// 考察找出来的两个阵营是不是真的超过人数
    for (int i = 0; i < numsSize; i++){
        int cur = nums[i]; // 遍历每个元素
        if(major1 == cur){
            cnt1++;
        } else if(major2 == cur){
            cnt2++;
        }
    }
// 如果人数有超过1/3就加入数组
    int *returned = NULL;
    if ((cnt1 * 3 > numsSize) && (cnt2 * 3 > numsSize)){
// 有两个
        returned = (int* )malloc(sizeof(int) * 2);
        returned[0] = major1;
        returned[1] = major2;
        *returnSize = 2; // 标记数量
    } else if ((cnt1 * 3 > numsSize) || (cnt2 * 3 > numsSize)){
// 有其中一个
        returned = (int* )malloc(sizeof(int) * 1);
        returned[0] = (cnt1 * 3 > numsSize)? major1: major2;
        *returnSize = 1; // 标记数量
    } else {
// 一个都没有
        *returnSize = 0; // 标记数量
    }
// 返回数组
    return returned;
}
```
