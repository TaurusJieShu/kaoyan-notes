
# 堆
堆的特点和构建方式：
- **完全二叉树**：堆是一种完全二叉树结构，具有连续的节点排列，避免了空间浪费。
- **堆顶值**：
    - **最大堆**（大根堆）：堆顶是最大值，任意节点的值都大于或等于其左右子节点。
    - **最小堆**（小根堆）：堆顶是最小值，任意节点的值都小于或等于其左右子节点。
- **堆的构建方式**：
    - **Inplace（就地构建）**：通过对非叶子节点进行下沉操作来构建堆。通常通过数组倒序遍历的方式，从最后一个非叶子节点开始，逐步调整堆的结构。
    - **Put（插入式构建）**：在新数组中逐个插入元素，并通过上浮操作将新插入的元素放到正确的位置，直到满足堆的性质。
- **Pop堆**：
    - 取出堆顶元素后，将堆的最后一个元素替换堆顶元素，并通过下沉操作调整堆，使其继续满足堆的性质。
```c
bool heap_comp(int *a, int *b){
	return *a > *b;
}

void swap(int* a, int* b) {
	int t = *a;
	*a = *b;
	*b = t;
	return;
}

void push(int num){
	int pos = ++heap_count;
	heap[pos] = num;
	while(pos > 1){
		if(heap_comp(heap[pos], heap[pos >> 1])) {
			swap(&heap[pos], &heap[pos >> 1]);
			pos >>= 1;
		} else
			break;
	}
	return;
}

  

int pop(){
	swap(heap[1], heap[heap_count]);
	heap_count --;
	int pos = 1;
	int compared;
	while((pos << 1) <= heap_count) {
		compared = pop << 1;
		if (heap_comp(heap[compared+1], heap[compared]) & compared < heap_count){
			compared ++;
		}
		if (heap_comp(heap[compared], heap[pos])) {
			swap(&heap[compared], &heap[pos]);
			pos = compared;
		}
	}
	
	return heap[heap_count + 1];
}
```

```c
#define maxn 100005

int **rec, *nx;

// --- 这里开始是堆的代码
int heap[maxn];
int heap_count;

bool heap_comp(int *first, int *second) {
    return rec[*first][nx[*first]] < rec[*second][nx[*second]];
}

void swap(int *first, int *second) {
    int temp = *second;
    *second = *first;
    *first = temp;
    return;
}

void push(int num) {
    int pos = ++heap_count; // 个数+1
    heap[pos] = num; // 把数据放在树末尾
    // 调整节点上移动
    while (pos > 1) {
        if (heap_comp(&heap[pos], &heap[pos >> 1])) { // self < father
            swap(&heap[pos], &heap[pos >> 1]); // 交换
            pos >>= 1; // 节点当前位置更新，直到节点不用更新，或pos = 1 为根节电
        } else
            break;
    }
    return;
}

void pop() {
    int top_num = 1;
    int now;
    swap(&heap[top_num], &heap[heap_count--]);
    while ((now = (top_num << 1)) <= heap_count) {
        if (heap_comp(&heap[now + 1], &heap[now]) && now < heap_count) now++;
        if (heap_comp(&heap[now], &heap[top_num])) {
            swap(&heap[top_num], &heap[now]);
            top_num = now;
        } else
            break;
    }
}

int top() { return heap[1]; }

// --- 这里开始是堆的代码

int *smallestRange(int **nums, int numsSize, int *numsColSize,
                   int *returnSize) {
    heap_count = 0;
    nx = (int *)malloc(sizeof(int) * numsSize);
    memset(nx, 0, sizeof(int) * numsSize);
    rec = nums;

    int rangeLeft = 0, rangeRight = 2147483647;
    int minValue = 0, maxValue = -2147483648;
    for (int i = 0; i < numsSize; ++i) { // 建立堆 并维护最大值
        push(i);
        maxValue = fmax(maxValue, nums[i][0]);
    }

    while (true) {
        int row = top(); // 获取最小元素
        pop();
        minValue = nums[row][nx[row]];
        
        if (maxValue - minValue < rangeRight - rangeLeft) {
            rangeLeft = minValue;
            rangeRight = maxValue;
        } // 如果找到比当前更小的距离，我就保存这个数据
        
        if (nx[row] == numsColSize[row] - 1) { // 当前要移动的是不是最后一个元素
            break;
        }
        ++nx[row];
        maxValue = fmax(maxValue, nums[row][nx[row]]);
        push(row);
    }
    int *ret = malloc(sizeof(int) * 2);
    ret[0] = rangeLeft, ret[1] = rangeRight;
    *returnSize = 2;
    return ret;
}
```
